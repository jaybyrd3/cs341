2     * This program is a calendar scheduler. It uses an 
3     * array of Day types, where each Day holds per-hour 
4     * scheduling data (whether an hour is scheduled, and 
5     * if it is what the appointment is scheduled).

13   import java.util.Scanner;
14   
15   public class Calendar {
16   
17   	/**
18   	 * This is the starting point for the calendar appointment scheduling program.
19   	 * It creates an array of Day objects, one for each day of the year. It then
20   	 * uses helper methods to print a menu and get user input. Then it accesses the
21   	 * day object selected by the user to do one of a handful of scheduling tasks.
22   	 *
23   	 * YOU SHOULD NOT MODIFY THIS METHOD!!
24   	 */
25   	public static void main(String[] args) {
26   		// Create the scanner
27   		Scanner scan = new Scanner(System.in);
28   
29   		// Allocate the array of Day references
30   		Day[] calendar = new Day[365];
31   		for (int i = 0; i < calendar.length; i++) {
32   			// Allocate each Day object
33   			calendar[i] = new Day();
34   		}
35   
36   		// These will hold user input of various types
37   		char menuSelection;
38   		int day, hour;
39   		String description;
40   
41   		// This loop repeatedly prints the menu and
42   		// handles the user selection until the user
43   		// chooses to end the program.
44   		do {
45   			// Print the menu and get the menu selection
46   			menuSelection = HandleInput.getMenu(scan);
47   
48   			// Act on whichever menu item the user selected
49   			if (menuSelection == 'c') {
50   				// Check for an appointment
51   
52   				// Prompt the user for the day of the appointment
53   				day = HandleInput.getDay(scan);
54   
55   				// Check if there are any appointments for the day
56   				if (calendar[day].checkDay()) {
57   					// If there are appontments for the day, then prompt for the hour
58   					hour = HandleInput.getHour(scan);
59   
60   					// Check if there is an appointment during the hour
61   					if (calendar[day].checkTime(hour)) {
62   						// Display the single appointment
63   						System.out.println(calendar[day].getDescription(hour));
64   					} else {
65   						System.out.println("No appointment scheduled");
66   					}
67   				} else {
68   					System.out.println("No appointments scheduled for that entire day");
69   				}
70   			} else if (menuSelection == 'a') {
71   				// Add a new appointment
72   
73   				// Get the day, hour and description from the user
74   				day = HandleInput.getDay(scan);
75   				hour = HandleInput.getHour(scan);
76   				description = HandleInput.getDescription(scan);
77   
78   				// Add the appointment, printing a message if there is
79   				// already an appointment at that time
80   				if (!calendar[day].addAppointment(hour, description)) {
81   					System.out.println("Time slot already has an appointment: " + calendar[day].getDescription(hour));
82   				}
83   			} else if (menuSelection == 'l') {
84   				// List all appointments for a day
85   
86   				// Get the day
87   				day = HandleInput.getDay(scan);
88   
89   				// Print the toString result
90   				System.out.println(calendar[day]);
91   			} else if (menuSelection == 'r') {
92   				// Remove an appointment
93   
94   				// Get the day and hour
95   				day = HandleInput.getDay(scan);
96   				hour = HandleInput.getHour(scan);
97   
98   				// Remove the appointment
99   				calendar[day].removeAppointment(hour);
100  			}
101  		} while (menuSelection != 'x');
102  
103  		// Print an exit message
104  		System.out.println("K thanks, bye!");
105  	}
106  
107  }
