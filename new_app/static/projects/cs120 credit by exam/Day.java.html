1    /**
2     * Class Day: This class is used to represent one day of a schedule. The data is
3     * an array of boolean values that represent whether an appointment is scheduled
4     * for a particular hour of the day. There is also a description that is
5     * associated with each hour of the day - saved in an array of Strings.
6     *
7     * Last Modified: January 15, 2023
8     * Author: Jimmy Byrd
9     */
10   
11   public class Day {
12   	// Attributes
13   
14   	// There are 24 hours in a day, so this
15   	// array should have 24 elements. If an
16   	// element is true, that means an appointment
17   	// is scheduled during that hour. If it is
18   	// false, then there is no appointment scheduled
19   	// during that hour.
20   	private boolean[] isBusy;
21   
22   	// This also should have 24 elements, one
23   	// for each hour of the day. If an appointment
24   	// is scheduled for an hour, then this array
25   	// should hold a String description of the
26   	// appointment.
27   	private String[] appointmentDescription;
28   
29   	/**
30   	 * Constructor for the Day class. This should allocate memory for the attribute
31   	 * arrays, and should initialize each hour to have no appointments.
32   	 * 
33   	 */
34   	public Day() {
35   		/* TODO - write this constructor */
36   		isBusy = new boolean[24];
37   		appointmentDescription = new String[24];
38   		//instantiate arrays
39   		for (int i = 0; i < 24; i++) {
40   			isBusy[i] = false;
41   			appointmentDescription[i] = "";
42   		}
43   	}
44   
45   	/**
46   	 * This method should return whether or not there is an appointment scheduled
47   	 * for a particular time of this day.
48   	 *
49   	 * @param hour
50   	 *            The hour during this day whose status is being checked.
51   	 * @return true if there is an appointment scheduled during the requested hour,
52   	 *         false if there is not.
53   	 */
54   	public boolean checkTime(int hour) {
55   		/* TODO - write this method */
56   		return isBusy[hour];
57   	}
58   
59   	/**
60   	 * This method should return whether or not there is an appointment scheduled at
61   	 * any hour of this day.
62   	 *
63   	 * @return true if there is an appointment scheduled during any hour of this
64   	 *         day, false if there are no appointments for this entire day.
65   	 */
66   	public boolean checkDay() {
67   		/* TODO - write this method */
68   		for (int i = 0; i < 24; i++) {
69   			if(isBusy[i]){
70   				return true; // returns true if any true in isBusy
71   			}
72   		}
73   		return false;
74   	}
75   
76   	/**
77   	 * This method should return the appointment description for the appointment at
78   	 * the given time of this day.
79   	 *
80   	 * @param hour
81   	 *            The hour during this day whose appointment description should be
82   	 *            returned.
83   	 * @return If an appointment is scheduled at the given hour, then the
84   	 *         appointment description should be returned, otherwise the text "No
85   	 *         appointment scheduled" should be returned.
86   	 */
87   	public String getDescription(int hour) {
88   		/* TODO - write this method */
89   		return appointmentDescription[hour]; //returns specific description for the given hour
90   	}
91   
92   	/**
93   	 * This method should add a new appointment to the given hour for this day. This
94   	 * method should modify the isBusy and appointmentDescription for the given day
95   	 * only if there is not already an appointment for the given hour.
96   	 *
97   	 * @param hour
98   	 *            The hour during this day for which an appointment should be made.
99   	 * @param description
100  	 *            The text description of the new appointment.
101  	 * @return false if there is already an appointment scheduled during the given
102  	 *         hour (the previous appointment information should not be modified),
103  	 *         and true if adding the appointment was successful (there was no
104  	 *         previous appointment).
105  	 */
106  	public boolean addAppointment(int hour, String description) {
107  		/* TODO - write this method */
108  		if(isBusy[hour]){
109  			return false;
110  		}
111  		else{
112  			isBusy[hour] = true;
113  			appointmentDescription[hour] = description;
114  			return true;
115  		}
116  	}
117  
118  	/**
119  	 * This method should remove an appointment for a given hour for this day. There
120  	 * should be no effect if there was not an appointment scheduled in the first
121  	 * place.
122  	 *
123  	 * @param hour
124  	 *            The hour during this day for which an appointment removed.
125  	 */
126  	public void removeAppointment(int hour) {
127  		/* TODO - write this method */
128  		if(isBusy[hour]){
129  			isBusy[hour]=false; // sets isBusy to false
130  			appointmentDescription[hour]=""; // clears description
131  		}
132  	}
133  
134  	/**
135  	 * This method should generate a String of all appointment times and
136  	 * descriptions for a given hour for this day. There should be no print
137  	 * statements in this method, only the code to create a String.
138  	 *
139  	 * @return A String whose text is one line for each appointment on this day.
140  	 *         Each line should have the appointment time and the appointment
141  	 *         description. See the program description for the exact formatting.
142  	 */
143  	public String toString() {
144  		/* TODO - write this method */
145  		//intellij suggested to use a string builder instead of repeatedly concatenating a blank string.
146  		StringBuilder msg= new StringBuilder();
147  		for (int i = 0; i < 24; i++) {
148  			if(isBusy[i]){
149  				// suggested way to concatenate the output string with formatting and 0 padding for proper 24hr time 00:00
150  				msg.append(String.format("%02d", i)).append(":00 - ").append(appointmentDescription[i]).append("\n");
151  			}
152  		}
153  		return msg.toString(); // suggested using the toString method with StringBuilder
154  	}
