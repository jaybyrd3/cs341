24   	 * This method will print a menu, and then read the user's menu selection and
25   	 * return that selection back to the caller. YOU SHOULD NOT MODIFY THIS METHOD,
26   	 * IT IS PROVIDED FOR YOU!!
27   	 * 
28   	 *
29   	 * @param scan
30   	 *            The Scanner variable must be passed from the caller, because it is
31   	 *            expected that only a single Scanner has been created for accessing
32   	 *            the console.
33   	 * @return The character that corresponds to the menu item selected by the user.
34   	 */
35   	public static char getMenu(Scanner scan) {
36   		String line;
37   		boolean done = false;
38   		char selection;
39   
40   		// Print out the menu items
41   		System.out.println("\nCalendar Menu:");
42   		System.out.println("\tc - Check calendar for appointments");
43   		System.out.println("\ta - Add a new calendar appointment");
44   		System.out.println("\tl - List appointments");
45   		System.out.println("\tr - Remove appointment");
46   		System.out.println("\tx - eXit the calendar program");
47   
48   		do {
49   			// Print out the prompt
50   			System.out.print("Enter choice: ");
51   
52   			// Get the user input, force it to lower case, and
53   			// get the first character
54   			line = scan.nextLine();
55   			line = line.toLowerCase();
56   			selection = line.charAt(0);
57   
58   			// Validate the input
59   			if (selection == 'c' || selection == 'a' || selection == 'l' || selection == 'r' || selection == 'x') {
60   				done = true;
61   			} else {
62   				// Print an error message if they entered a bad input
63   				System.out.println("ERROR: bad menu item, please select from the list above");
64   			}
65   		} while (!done);
66   		System.out.println();
67   
68   		// return the character of the selected menu item
69   		return selection;
70   	}

94   	 * This method will verify whether a value is a valid day value. Days are
95   	 * integers 0 through 364, where 0 corresponds to Jan 1, and 365 corresponds to
96   	 * Dec 31.
97   	 *
98   	 * @param day
99   	 *            The day value that is being checked.
100  	 * @return true if the day is a valid value, false if it is not.

102  	public static boolean verifyDay(int day) {
103  		/* TODO - write this method */
104  		//1 liner operator expression! :)
105  		return (day>=0&&day<=364);

117  	public static boolean verifyHour(int hour) {
118  		/* TODO - write this method */

123  	/**
124  	 * This method will prompt the user to enter a day. Their input must be validity
125  	 * checked by using the verifyDay() method from above. If a bad value is
126  	 * entered, an error message should be printed and the user should be prompted
127  	 * again (and until they enter a correct value). The day they enter must then be
128  	 * returned to the caller.
129  	 *
130  	 * @param scan
131  	 *            The Scanner variable for accessing console input.
132  	 * @return The day entered by the user. Must be a value between 0 and 364.
133  	 */
134  	public static int getDay(Scanner scan) {
135  		/* TODO - write this method */
136  		int day;
137  		while (true) {
138  			System.out.print("What day?");
139  			if (scan.hasNextInt()) {
140  				day = scan.nextInt();
141  				scan.nextLine(); //clear buffer
142  				if (verifyDay(day)) {
143  					return day;
144  				} else {
145  					System.out.println("ERROR: day must be between 0 and 364");
146  				}
147  			} else {
148  				System.out.println("ERROR: day must be between 0 and 364");
149  				scan.nextLine(); //clear buffer
150  			}
151  		}
152  	}
