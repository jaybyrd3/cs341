22   /**
23   	GetValidInput gets a valid input from the user.
24    	It will accept all inputs without exceptions.
25   	I would rather use try except instead of do while.
26   	I did not see that cover in cs 120, so I used the do while similar to the Calendar class.
27    	@param prompt takes a string prompt to determine case evaluation
28    	@param minVal take a minimum value to verify
29    	@param maxVal takes a maximum value to verify
30    	@reutrn returns a valid input value
31   */
32    	private int getValidInput(String prompt, int minVal, int maxVal) {
33   		int inputVal;
34   		Scanner scanner = new Scanner(System.in);
35   		do {
36   			System.out.print(prompt);
37   			if (scanner.hasNextInt()) {
38   				inputVal = scanner.nextInt();
39   				if(inputVal < minVal || inputVal > maxVal) {
40   					if (prompt.equals("Is this a leap year [1=yes / 0=no]?")) {
41   						System.out.println("Error: Enter either 1 for \"yes\" or 0 for \"no\"");
42   					} else {
43   						System.out.println("Error: Enter a value between " + minVal + " and " + maxVal + " (inclusive).");
44   					}
45   					inputVal = Integer.MIN_VALUE;//min int value instead of 0.
46   				}
47   			}else{
48   				if (prompt.equals("Is this a leap year [1=yes / 0=no]?")) {
49   					System.out.println("Error: Enter either 1 for \"yes\" or 0 for \"no\"");
50   				} else {
51   					System.out.println("Error: Enter a value between " + minVal + " and " + maxVal + " (inclusive).");
52   				}
53   				scanner.next();
54   				inputVal = Integer.MIN_VALUE;//min int value instead of 0.
55   			}
56   		} while (inputVal == Integer.MIN_VALUE);//min int value instead of 0.
57   		return inputVal;
58   	}
59   /**
60   	Gets the starting day of the month.
61   */
62    	public String getStartingDay() {
63   		int startDay = getValidInput("1=Su., 2=Mo., 3=Tu., 4=We., 5=Th., 6=Fr., 7=Sa.\nEnter the day of the week [1-7]:", 1, 7);
64   		return weekdays[startDay - 1];
65   	}
66   /**
67   	 Gets the month to create the calendar for.
68   */
69   	public String getMonth() {
70   		int month = getValidInput("1=Jan., 2=Feb., ..., 11=Nov., 12=Dec.\nEnter the month [1-12]:", 1, 12);
71   		return months[month - 1];
72   	}

74   /**
75     	Returns boolean value for leap year.
76   */
77   	public boolean isLeapYear() {
78   		int leapYear = getValidInput("Is this a leap year [1=yes / 0=no]?", 0, 1);
79   		return leapYear == 1;//returns true if leapYear
80   	}
81   /**
82   	Displays the proper calendar with starting date and selected month and leap year if applicable.
83   	I used the printf method as discussed on the Zoom meeting.
84   */
85   	public void displayCalendar() {
86   		String startDay = getStartingDay();
87   		String month = getMonth();
88   		boolean leapYear = false;
89   		if(month.equals("February")){
90   			leapYear = isLeapYear();
91   		}
92   		if (leapYear) {
93   			monthDays[1] = 29;
94   		}
95   		System.out.println(month);
96   		System.out.println("Su Mo Tu We Th Fr Sa");
97   		int startDayIndex = 0;
98   		for (int i = 0; i < weekdays.length; i++) {
99   			if (weekdays[i].equals(startDay)) {
100  				startDayIndex = i;
101  				break;
102  			}
103  		}
104  		int monthIndex = 0;
105  		for (int i = 0; i < months.length; i++) {
106  			if (months[i].equals(month)) {
107  				monthIndex = i;
108  				break;
109  			}
110  		}
111  		int daysInMonth = monthDays[monthIndex];
112  		for (int i = 0; i < startDayIndex; i++) {
113  			System.out.print("   ");
114  		}
115  		for (int i = 1; i <= daysInMonth; i++) {
116  			System.out.printf("%2d ", i);//printf was very useful here.
117  			if ((i + startDayIndex) % 7 == 0) {
118  				System.out.println();
119  			}
120  		}
121  		System.out.println();
122  	}
